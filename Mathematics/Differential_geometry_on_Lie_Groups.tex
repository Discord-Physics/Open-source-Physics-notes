\section{Lie Groups}

\definition{Definition}{
    A \textbf{Lie group} $G$ is a group which is at the same time a differentiable manifold such that the group operations:
    \[
        G\times G\xrightarrow{\text{into}}G\qquad\text{i.e.}\qquad(x,y)\longmapsto x\,\cdot\,y
    \]
    and
    \[
        G\longrightarrow G\qquad\text{i.e.}\qquad x\longmapsto x^{-1}
    \]
    are $C^\infty$-maps.
}
